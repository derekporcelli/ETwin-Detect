# 1 - put your adapter in monitor mode (root required)
sudo ip link set wlan0 down
sudo iw wlan0 set monitor control
sudo ip link set wlan0 up

# 2 - create the baseline while the airspace is clean
sudo python3 evil_twin.py --iface wlan0 --profile --seconds 90

# 3 - later / on a different host, start monitoring
sudo python3 evil_twin.py --iface wlan0 --monitor
